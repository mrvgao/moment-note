''' 

Build a new create app command. Will create apps.some_app automatically. 

@Author Minchiuan Gao <2016-4-20> 

''' 

 

from django.core.management.base import BaseCommand 

import os.path 

from django.core.management.base import CommandError 

from importlib import import_module 

 

APP_BASE = 'apps/' 

 

def create_app(app_name, base_dir='apps'): 

    NECESSEARY_MODULES = ['apis.py', 'models.py', 'serializers.py', 'services.py', 'urls.py'] 

    NECESSARRY_DIRS = ['tests', 'utils'] 

 

 

class Command(BaseCommand): 

    def handle(self, **options): 

        app_name = options.pop('name') 

        if(os.path.isfile(APP_BASE+app_name)): 

            print('app is already existed') 

            raise CommandError("%s already existed " % app_name) 

        else: 

            try: 

                import_module(app_name) 

            except ImportError: 

                create_app(app_name) 

            else: 

                raise CommandError('%r conficts with the name of python module' % app_name)
